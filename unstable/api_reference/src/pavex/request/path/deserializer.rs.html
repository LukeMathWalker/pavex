186struct MapDeserializer<'de, 'request> {
187    params: &'de [(&'request str, Cow<'request, str>)],
188    key: Option<Key<'request>>,
189    value: Option<&'de Cow<'request, str>>,
190}
191
192impl<'request> MapAccess<'request> for MapDeserializer<'_, 'request> {
193    type Error = PathDeserializationError;
194
195    fn next_key_seed<K>(&mut self, seed: K) -> Result<Option<K::Value>, Self::Error>
196    where
197        K: DeserializeSeed<'request>,
198    {
199        match self.params.split_first() {
200            Some(((key, value), tail)) => {
201                self.value = Some(value);
202                self.params = tail;
203                self.key = Some(Key(key));
204                seed.deserialize(KeyDeserializer {
205                    key: Cow::Borrowed(key),
206                })
207                .map(Some)
208            }
209            None => Ok(None),
210        }
211    }
212
213    fn next_value_seed<V>(&mut self, seed: V) -> Result<V::Value, Self::Error>
214    where
215        V: DeserializeSeed<'request>,
216    {
217        match self.value.take() {
218            Some(value) => seed.deserialize(ValueDeserializer {
219                key: self.key.take(),
220                value: value.clone(),
221            }),
222            None => Err(PathDeserializationError::custom("value is missing")),
223        }
224    }
225}

231macro_rules! parse_key {
232    ($trait_fn:ident) => {
233        fn $trait_fn<V>(self, visitor: V) -> Result<V::Value, Self::Error>
234        where
235            V: Visitor<'de>,
236        {
237            visitor.visit_str(&self.key)
238        }
239    };

242impl<'de> Deserializer<'de> for KeyDeserializer<'de> {
243    type Error = PathDeserializationError;
244
245    parse_key!(deserialize_identifier);
246    parse_key!(deserialize_str);
247    parse_key!(deserialize_string);

249    fn deserialize_any<V>(self, _visitor: V) -> Result<V::Value, Self::Error>

254    }

256    forward_to_deserialize_any! {

259        tuple_struct map newtype_struct struct enum ignored_any
260    }

263macro_rules! parse_value {
264    ($trait_fn:ident, $visit_fn:ident, $ty:literal) => {
265        fn $trait_fn<V>(mut self, visitor: V) -> Result<V::Value, Self::Error>
266        where
267            V: Visitor<'request>,
268        {
269            let v = self.value.parse().map_err(|_| {
270                if let Some(key) = self.key.take() {
271                    let kind = ErrorKind::ParseErrorAtKey {
272                        key: key.0.to_string(),
273                        value: self.value.to_string(),
274                        expected_type: $ty,
275                    };
276                    PathDeserializationError::new(kind)
277                } else {
278                    PathDeserializationError::new(ErrorKind::ParseError {
279                        value: self.value.to_string(),
280                        expected_type: $ty,
281                    })
282                }
283            })?;
284            visitor.$visit_fn(v)
285        }
286    };
287}
