21/// A handle to dispatch incoming connections to a worker thread.
22pub(super) struct WorkerHandle {
23    connection_outbox: tokio::sync::mpsc::Sender<ConnectionMessage>,
24    // We use an unbounded channel because we want to be able to send a shutdown command
25    // synchronously.
26    shutdown_outbox: tokio::sync::mpsc::UnboundedSender<ShutdownWorkerCommand>,
27    id: usize,
28}
29
30impl WorkerHandle {
31    /// Dispatch a connection to the worker thread.
32    pub(super) fn dispatch(
33        &self,
34        connection: ConnectionMessage,
35    ) -> Result<(), TrySendError<ConnectionMessage>> {
36        self.connection_outbox.try_send(connection)
37    }
38
39    /// Get the worker's ID.
40    pub(super) fn id(&self) -> usize {
41        self.id
42    }
43
44    /// Shutdown the worker thread.
45    ///
46    /// # Implementation notes
47    ///
48    /// We use a sync function to ensure that the shutdown command is enqueued immediately,
49    /// even if the returned future is never polled.
50    pub(super) fn shutdown(self, mode: ShutdownMode) -> impl Future<Output = ()> {
51        let (completion_notifier, completion) = tokio::sync::oneshot::channel();
52        let sent = self
53            .shutdown_outbox
54            .send(ShutdownWorkerCommand {
55                completion_notifier,
56                mode,
57            })
58            .is_ok();
59        async move {
60            // What if sending fails?
61            // It only happens if the other end of the channel has already been dropped, which
62            // implies that the worker thread has already shut downâ€”nothing to do!
63            if sent {
64                let _ = completion.await;
65            }
66        }
67    }
68}

72    mode: ShutdownMode,

76/// A worker thread that handles incoming connections.
77pub(super) struct Worker<HandlerFuture, ApplicationState> {
78    connection_inbox: tokio::sync::mpsc::Receiver<ConnectionMessage>,
79    shutdown_inbox: tokio::sync::mpsc::UnboundedReceiver<ShutdownWorkerCommand>,
80    handler: fn(
81        http::Request<hyper::body::Incoming>,
82        Option<ConnectionInfo>,
83        ApplicationState,
84    ) -> HandlerFuture,
85    application_state: ApplicationState,
86    id: usize,
87    shutdown_coordinator: GracefulShutdown,
88}
89
90impl<HandlerFuture, ApplicationState> Worker<HandlerFuture, ApplicationState>
91where
92    HandlerFuture: Future<Output = crate::Response> + 'static,
93    ApplicationState: Clone + Send + Sync + 'static,
94{
95    /// Configure a new worker without spawning it.
96    ///
97    /// `max_queue_length` is the maximum number of connections that can be queued up for this
98    /// worker.
99    pub(super) fn new(
100        id: usize,
101        max_queue_length: usize,
102        handler: fn(
103            http::Request<hyper::body::Incoming>,
104            Option<ConnectionInfo>,
105            ApplicationState,
106        ) -> HandlerFuture,
107        application_state: ApplicationState,
108    ) -> (Self, WorkerHandle) {
109        let (connection_outbox, connection_inbox) = tokio::sync::mpsc::channel(max_queue_length);
110        let (shutdown_outbox, shutdown_inbox) = tokio::sync::mpsc::unbounded_channel();
111        let self_ = Self {
112            connection_inbox,
113            shutdown_inbox,
114            handler,
115            application_state,
116            id,
117            shutdown_coordinator: GracefulShutdown::new(),
118        };
119        let handle = WorkerHandle {
120            connection_outbox,
121            shutdown_outbox,
122            id,
123        };
124        (self_, handle)
125    }
126
127    /// Spawn a thread and run the worker there, using a single-threaded executor that can
128    /// handle !Send futures.
129    pub(super) fn spawn(self) -> Result<thread::JoinHandle<()>, anyhow::Error> {
130        let id = self.id;
131        let name = || format!("pavex-worker-{id}");
132        thread::Builder::new()
133            .name(name())
134            .spawn(move || {
135                let runtime = tokio::runtime::Builder::new_current_thread()
136                    .enable_all()
137                    .build()
138                    // This can only fail if `tokio` fails to create an IO stack for the
139                    // current thread executor, which is highly unlikely to happen
140                    // and probably indicates a deeper issue that we can't recover from.
141                    .expect("Failed to build single-threaded Tokio runtime for worker thread");
142
143                // Use a `LocalSet` to allow the worker to spawn !Send futures.
144                let local = tokio::task::LocalSet::new();
145                local.block_on(&runtime, self.run());
146            })
147            .with_context(|| format!("Failed to spawn worker thread `{}`", name()))
148    }
149
150    /// Run the worker: wait for incoming connections and handle them.
151    async fn run(self) {
152        let Self {
153            mut connection_inbox,
154            mut shutdown_inbox,
155            handler,
156            application_state,
157            id,
158            shutdown_coordinator,
159        } = self;
160        'event_loop: loop {
161            let message =
162                poll_fn(|cx| Self::poll_inboxes(cx, &mut shutdown_inbox, &mut connection_inbox))
163                    .await;
164            match message {
165                WorkerInboxMessage::Connection(connection) => {
166                    Self::handle_connection(
167                        connection,
168                        handler,
169                        application_state.clone(),
170                        &shutdown_coordinator,
171                    );
172                }
173                WorkerInboxMessage::Shutdown(shutdown) => {
174                    let ShutdownWorkerCommand {
175                        completion_notifier,
176                        mode,
177                    } = shutdown;
178                    match mode {
179                        ShutdownMode::Graceful { timeout } => {
180                            // Stop accepting new connections.
181                            connection_inbox.close();
182
183                            // Kick-off work for all pending connections.
184                            while let Some(connection) = connection_inbox.recv().await {
185                                Self::handle_connection(
186                                    connection,
187                                    handler,
188                                    application_state.clone(),
189                                    &shutdown_coordinator,
190                                );
191                            }
192
193                            // Wait for all live connections to be closed or for the timeout to expire.
194                            let _ = tokio::time::timeout(timeout, shutdown_coordinator.shutdown())
195                                .await;
196                        }
197                        ShutdownMode::Forced => {}
198                    }
199                    let _ = completion_notifier.send(());
200                    break 'event_loop;
201                }
202            }
203        }
204        tracing::info!(worker_id = id, "Worker shut down");
205    }
206
207    fn handle_connection(
208        connection_message: ConnectionMessage,
209        handler: fn(
210            http::Request<hyper::body::Incoming>,
211            Option<ConnectionInfo>,
212            ApplicationState,
213        ) -> HandlerFuture,
214        application_state: ApplicationState,
215        shutdown_coordinator: &GracefulShutdown,
216    ) {
217        let ConnectionMessage {
218            connection,
219            peer_addr,
220        } = connection_message;
221        // A tiny bit of glue to adapt our handler to hyper's service interface.
222        let handler = hyper::service::service_fn(move |request| {
223            let state = application_state.clone();
224
225            async move {
226                let handler = (handler)(request, Some(ConnectionInfo { peer_addr }), state);
227                let response = handler.await;
228                let response = hyper::Response::from(response);
229                Ok::<_, hyper::Error>(response)
230            }
231        });
232        // TODO: expose all the config options for `auto::Builder` through the top-level
233        //   `ServerConfiguration` object.
234        let builder = hyper_util::server::conn::auto::Builder::new(LocalExec);
235        let connection = TokioIo::new(connection);
236        let connection_future =
237            shutdown_coordinator.watch(builder.serve_connection(connection, handler).into_owned());
238        tokio::task::spawn_local(async move {
239            if let Err(e) = connection_future.await {
240                log_error!(*e, level: tracing::Level::WARN, "Failed to serve an incoming connection");
241            }
242        });
243    }
244
245    /// Check if there is work to be done.
246    fn poll_inboxes(
247        cx: &mut std::task::Context<'_>,
248        shutdown_inbox: &mut tokio::sync::mpsc::UnboundedReceiver<ShutdownWorkerCommand>,
249        connection_inbox: &mut tokio::sync::mpsc::Receiver<ConnectionMessage>,
250    ) -> Poll<WorkerInboxMessage> {
251        // Order matters here: we want to prioritize shutdown messages over incoming connections.
252        if let Poll::Ready(Some(message)) = shutdown_inbox.poll_recv(cx) {
253            return Poll::Ready(message.into());
254        }
255        if let Poll::Ready(Some(message)) = connection_inbox.poll_recv(cx) {
256            return Poll::Ready(message.into());
257        }
258        Poll::Pending
259    }
260}

279/// HTTP2 requires `hyper` to be able to spawn tasks, therefore we need to pass to `hyper`'s
280/// `Server` an executor and a way to perform the spawning.
281///
282/// We use `spawn_local` since we want each worker thread to be able to spawn !Send futures.
283#[derive(Clone, Copy, Debug)]
284struct LocalExec;
285
286impl<F> hyper::rt::Executor<F> for LocalExec
287where
288    F: Future + 'static, // no `Send`
289{
290    fn execute(&self, fut: F) {
291        // This will spawn into the currently running `LocalSet`.
292        tokio::task::spawn_local(fut);
293    }
